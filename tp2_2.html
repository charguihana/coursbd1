📘 Correction TP2 — Poste Frontalier

Q1 : Ajouter contrainte Genre

ALTER TABLE voyageur ADD CONSTRAINT v CHECK(Genre='M' OR Genre='F');
CHECK permet de contrôler les valeurs autorisées dans la colonne Genre.

Q2 : Modifier type Frontière

ALTER TABLE poste CHANGE front Frontiere CHAR(2);
CHANGE modifie le nom ou le type d’un attribut.

R1 : Afficher postes commerciaux

SELECT CodeP, Nom, Gouvernorat FROM poste WHERE Categorie='C';
WHERE filtre les postes ayant catégorie commerciale.

R2 : Gouvernorat contenant 'd'

SELECT * FROM poste WHERE Gouvernorat LIKE '%d%';
LIKE permet la recherche partielle de texte.

R3 : Voyageurs sortis avant Février 2024

SELECT v.NomPrenom,v.Nationalite,v.Genre FROM voyageur v, passage p WHERE v.IdVoy=p.IdVoy AND YEAR(p.Date)=2024 AND MONTH(p.Date)<2;
Jointure entre voyageur et passage puis filtrage date.

R4 : Nombre voyageurs sortants

SELECT COUNT(*) nb_sortant FROM passage WHERE Acces='S';
COUNT(*) calcule le nombre total des sorties.

R5 : Ajouter 5000 m²

UPDATE poste SET superficie = superficie + 5000 WHERE superficie BETWEEN 10000 AND 50000;
UPDATE modifie les valeurs existantes selon condition.

R6 : Trier voyageurs

SELECT * FROM voyageur ORDER BY NomPrenom ASC;
ORDER BY permet le tri alphabétique.
